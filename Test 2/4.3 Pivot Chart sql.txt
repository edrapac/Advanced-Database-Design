With T1 as (SELECT C.CUSTOMER_ID CID, C.CUSTOMER.FIRST_NAME cfname,C.CUSTOMER.LAST_NAME clname
       ,C.CUSTOMER.GENDER cg,C.ADDRESS ADDRESS,
       E.EMPLOYEE_ID eid,E.EMPLOYEE.LAST_NAME elname,
SUBSTR(CO.DATE_CREATED,4,3) as MonthOrdered,SUBSTR(CO.DATE_CREATED,8,2) as YearCreated,
       PO.PAINTING_QTY QtyOrdered,(P.PRICE*PO.PAINTING_QTY) Sale, P.TITLE Title, S.STYLE_NAME sname
FROM O_Employee E

JOIN CUSTOMER_ORDER CO ON CO.EMPLOYEE_ID = E.EMPLOYEE_ID
JOIN O_CUSTOMER C on CO.CUSTOMER_ID = C.CUSTOMER_ID
JOIN PAINTING_ORDER PO on CO.ORDER_ID = PO.ORDER_ID
JOIN PAINTING P on PO.PAINTING_ID = P.PAINTING_ID
JOIN PAINTING_STYLE PS on P.PAINTING_ID = PS.PAINTING_ID
JOIN STYLES S on PS.STYLE_ID = S.STYLE_ID)

SELECT T1.CID, T1.cfname, T1.clname, T1.cg, A.*, T1.eid, T1.elname, T1.MonthOrdered, T1.YearCreated, T1.QtyOrdered, T1.Sale, T1.Title, T1.sname
FROM T1,TABLE(T1.ADDRESS) A
